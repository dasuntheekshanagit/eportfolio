Including Plots

New Corona Cases in Sri Lanka

ggplot(data = corona_table) +
  geom_point(aes(x = as.Date(date), y = new_cases))+
  geom_smooth(aes(x = as.Date(date), y = new_cases),method = 'loess',se = FALSE,formula = 'y~x')+
  labs(title='New Corona Cases in Sri Lanka',subtitle = 'Period: 2020-01-27  ~  2021-08-05',caption= 'Data Source: https://ourworldindata.org/coronavirus/country/sri-lanka',y='New Cases',x='Date')+
  theme(axis.text.x = element_text(angle = 90))

Total Corona Cases in Sri Lanka

ggplot(data = corona_table) +
  geom_point(mapping = aes(x = as.Date(date), y = total_cases),color='RED')+
  geom_smooth(aes(x = as.Date(date), y = total_cases),method = 'loess',se = FALSE,formula = 'y~x')+

  theme(axis.text.x = element_text(angle = 90))
